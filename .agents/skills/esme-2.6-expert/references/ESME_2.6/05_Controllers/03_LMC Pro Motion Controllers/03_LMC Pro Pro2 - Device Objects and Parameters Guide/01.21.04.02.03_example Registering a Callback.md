# Example: Registering a Callback

## Example

```
PROGRAM SR_RegisterUserCallback
VAR
   xAlreadyRegistered : BOOL := FALSE;
   fbUserCallback : FB_UserCallback;
   ifPBSlaveEventHandler : IF_PROFIBUS_DPV1_Slave := PROFIBUS_DPV1_Slave;
   xResult1 : BOOL := FALSE;
   xResult2 : BOOL := FALSE;
   xResult3 : BOOL := FALSE;
   udiIndex : UDINT := 0;
END_VAR
IF (xAlreadyRegistered=FALSE) THEN
   // Register for callback when async read is requested
   xResult1 := ifPBSlaveEventHandler.RegisterAsyncRead(fbUserCallback);
   // Set values of global array that is returned from FB_UserCallback.AsyncRead()
   // when master requests slot 1, index 120
   FOR (udiIndex:=0) TO (Gc_C1AsyncRead_MaxDataLength-1) DO
      G_abyDataTransfer[udiIndex] := UDINT_TO_BYTE(udiIndex);
   END_FOR
   // Register for callback when async write is requested
   xResult2 := ifPBSlaveEventHandler.RegisterAsyncWrite(fbUserCallback);
   // Register for callback when alarm acknowledge arrives
   xResult3 := ifPBSlaveEventHandler.RegisterAsyncAlarmAck(fbUserCallback);
   xAlreadyRegistered := TRUE;
END_IF
```

END_VAR
