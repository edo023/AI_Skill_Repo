### Query the current state of a message queue

> ```
> hQueue : MBM.HANDLE;
> eError : MBM.ERROR;
> xQuery : BOOL;
>
> udiCount : UDINT;
> xEmpty : BOOL;
>
> IF xQuery THEN
>     udiCount := TO_UDINT(MBM.XChgMsgLeft(hXChg:=hQueue, peError:=ADR(eError));
>     xEmpty := MBM.XChgIsEmpty(hXChg:=hQueue, peError:=ADR(eError)
>     xQuery := FALSE;
> END_IF
> ```
